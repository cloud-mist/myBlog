func getDivisors(n int) []int {
	res := make([]int, 0)

	// 约数是成对出现的,添加一个，再添加除的的结果的那个
	for i := 1; i <= n/i; i++ {
		if n%i == 0 {
			res = append(res, i)
			if n/i != i { // 不重复添加
				res = append(res, n/i)
			}
		}
	}

	sort.Ints(res)
	return res
}

func gcd(a, b int) int {
	for a != b {
		if a > b {
			a -= b
		} else {
			b -= a
		}
	}
	return a
}

func gcd(a, b int) int {
